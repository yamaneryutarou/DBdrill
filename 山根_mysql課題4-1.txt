	CREATE TABLE #テーブル作成

 		(
 			author_id CHAR(5) PRIMARY KEY,
 			name VARCHAR(30),
 			name_kana VARCHAR(100) ,
 			birth DATE
 		)
	;
	
	
2
	CREATE TABLE #テーブル作成
		order_desc #注文明細テーブル
		(
			po_id INT NOT NULL,
			p_id CHAR(10) NOT NULL,
			quantity INT,
			PRIMARY KEY (po_id,p_id)
		)
	;


3
	CREATE TABLE #テーブル作成
		sales #月間売り上げテーブル
		(
			s_id CHAR(5) NOT NULL,
			s_date CHAR(7) NOT NULL,
			s_values INT,
			PRIMARY KEY (s_id,s_date)
		)
	;
	
	
4
	CREATE TABLE #テーブル作成
		books #書籍情報テーブル
		(
			isbn CHAR(17) NOT NULL PRIMARY KEY,
			title VARCHAR(30),
			price INT,
			publish VARCHAR(30),
			publish_date DATE,
			category_id CHAR(5)
		)
	;
