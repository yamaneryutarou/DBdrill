	SELECT #取得
		* #全件
	FROM
		quest #アンケート回答テーブル
	WHERE
		sex = '女' #性別が女性
	  AND age between 20 AND 29 #年齢が20歳代
	;
	
	
2
	SELECT #取得
		l_name, #氏
		f_name, #名
		email #メールアドレス
	FROM
		usr #ユーザーテーブル
	WHERE
		prefecture = '東京都' #出身が東京都
	  AND
	    email LIKE '%examples.com' #メールアドレスのドメインが(examples.com)
	;


3
	SELECT #取得
		answer2 #本文の感想2
	FROM
		quest #アンケート回答テーブル
	WHERE
		answer2 <> '' #本文の感想2が空欄ではない
	 AND
	  answer2 IS NOT NULL #本文の感想2がNULLではない
	;

	WHERE
		l_name = '山田' #性が山田
	 AND
