	SELECT #取得
		prefecture, #出身地
		AVG(age) #年齢の平均
	FROM
		quest #アンケート回答テーブル
	GROUP BY
		prefecture #出身地でグループ化
	HAVING
		AVG(age) BETWEEN 35 AND 49 #35歳以上50歳未満
	;


2
	SELECT #取得
		prefecture, #出身地
		MAX(age) #最高年齢
	FROM
		quest #アンケート回答テーブル
	WHERE
		sex = '男'
	GROUP BY
		prefecture #出身地でグループ化
	HAVING
		MAX(age) > 60 #60歳より大きい
	;


3
	SELECT #取得
		author_id, #著者ID
		COUNT(isbn) AS 書籍数 #書籍数
	FROM
		author_books #著者-書籍情報テーブル
	GROUP BY
		author_id #著者IDでグループ化
	HAVING
		COUNT(isbn) >= 3 #書籍数が3冊以上
	;
	
	
4
	SELECT #取得
		publish, #出版社
		category_id, #分類ID
		COUNT(*) AS 書籍数 #書籍数
	FROM
		books #本テーブル
	GROUP BY
		publish, category_id #著者ID,分類IDでグループ化
	HAVING
		COUNT(*) < 3 #書籍数が3冊未満
	;


5
	SELECT #取得
		depart_id, #所属部署コード
		COUNT(*)  #数
	FROM
		employee #社員テーブル
	WHERE
		sex = 2 #性別が女性
	GROUP BY #「ORDERではなくGROUP」
		depart_id #所属部署コードでグループ化
	HAVING #「WHEREではなくHAVING」
		COUNT(*) >= 3 #3人以上
	;


6
	SELECT #取得
		referer, #リンク元のURL
		COUNT(*) #アクセス数
	FROM
		access_log #アクセス記録テーブル
	WHERE
		access_date >= '2013-01-01' #
	GROUP BY
		referer
	HAVING
		COUNT(*) < 5
	ORDER BY
		COUNT(*) DESC
	;
