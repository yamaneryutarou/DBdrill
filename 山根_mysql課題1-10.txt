	SELECT #取得
		p_name AS 商品名, #商品名
		price AS 価格 #価格
	FROM
		product #商品テーブル
	ORDER BY
		price ASC #価格が安い順
	;


2
	SELECT #取得
		prefecture AS 都道府県名, #出身
		AVG(age) AS 平均年齢
	FROM
		quest #アンケート回答テーブル
	GROUP BY
		prefecture #出身地でグループ化
	;


3
	SELECT #取得
		prefecture AS 都道府県名, #出身地
		COUNT(user_id) AS ユーザー数 #ユーザー数
	FROM
		usr #ユーザーテーブル
	GROUP BY
		prefecture #出身地でグループ化
	;


4
	SELECT #取得
		publish AS 出版社, #出版社
		AVG(price) AS 価格平均 #平均価格
	FROM
		books #本テーブル
	GROUP BY
		publish #出版社でグループ化
	;


5
	SELECT #取得
		prefecture, #出身地
		sex, #性別
		#「ageは不要」
		AVG(answer1) AS 評価平均 #「ISではなくAS」評価平均
	FROM
		quest #アンケート回答テーブル
	GROUP BY
		prefecture,sex #出身地,性別,年齢でグループ化
	;
