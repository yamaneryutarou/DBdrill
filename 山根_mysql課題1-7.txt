	SELECT #取得
		title, #タイトル
		price #価格
	FROM
		books #本テーブル
	WHERE
		price BETWEEN 2500 AND 3500 #価格が2500円〜3500円
	ORDER BY
		price ASC #価格が安い順
	;
	
	
2
	SELECT #取得
		l_name, #氏
		f_name, #名
		prefecture #出身地
	FROM
		usr #ユーザーテーブル
	WHERE
		prefecture IN ('東京都','千葉県','神奈川県') #(東京都,千葉県,神奈川県に住んでいる人で絞り込む)
	ORDER BY
		l_name_kana ASC, #姓(カナ)昇順
		f_name_kana ASC #名(カナ)昇順
	;


3
	SELECT #取得
		* #全件
	FROM
		rental #貸し出し記録テーブル
	WHERE
		returned = 0 #貸し出し中
	 AND
	 	rental_date < '2012-12-01' #貸出日より前
	ORDER BY
		rental_date DESC #貸出日が新しい順
	;

	SELECT #取得
		* #全件
	FROM
		access_log #アクセス記録テーブル
	WHERE
		access_date BETWEEN '2013-01-01' AND '2013-01-31'#「BETWEENを使用すること」
	ORDER BY
		referer DESC, #「DESCが不足」リンク元のURLが降順
		ip_address DESC #IPアドレスが降順
	;
