	SELECT #取得
		e.s_id, #社員コード
		t.r_date #日付
	FROM
		employee AS e #社員テーブル
	INNER JOIN
		time_card AS t #タイムカードテーブル
	ON
		e.s_id = t.s_id #店舗コード
	WHERE
		t.r_date BETWEEN '2012-12-01' AND '2012-12-31' #2012年12月
	ORDER BY
		e.s_id ASC, #社員コード 昇順
		t.r_date ASC #日付 昇順
	;
	
	
2
	SELECT #取得
		sh.s_name, #店舗名
		sl.s_value #売上高
	FROM
		shop AS sh #店舗テーブル
	INNER JOIN
		sales AS sl #月間売り上げテーブル
	ON
		sh.s_id = sl.s_id #店舗コード
	WHERE
		sl.s_date = '2012-12' #2012年12月
	ORDER BY
		sl.s_value DESC #売上高 降順
	;
	


3
	SELECT #取得
		sh.s_name, #店舗名
		sl.s_value #売上高
	FROM
		shop AS sh #店舗テーブル
	INNER JOIN
		sales AS sl #月間売り上げテーブル
	ON
		sh.s_id = sl.s_id #店舗コード
	WHERE
		sl.s_date BETWEEN '2012-01' AND '2012-12' #2012年
	GROUP BY
		sh.s_id, #店舗コード
		sh.s_name #店舗名
	ORDER BY
		SUM(sl.s_value) ASC #売上高 昇順
	;


4
	SELECT #取得
		CONCAT(u.l_name,u.f_name), #氏名
		COUNT(r.id) #貸し出し数
	FROM
		rental AS r #貸し出し記録テーブル
	INNER JOIN
		usr AS u #ユーザテーブル
	ON
		r.user_id = u.user_id #利用者コード
	WHERE
		r.returned = 0 #貸し出し中
	GROUP BY
		u.l_name, #氏
		u.f_name, #名
		r.user_id #利用者コード
	HAVING
		COUNT(r.id) > 0 #貸し出し数0より多い
	ORDER BY
		COUNT(r.id) DESC #貸し出し数の多い順
	;
